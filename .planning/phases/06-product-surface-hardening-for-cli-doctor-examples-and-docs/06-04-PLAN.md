```rust
#[test]
fn smoke_compile_train_rejects_non_mlp() {
    // Write a minimal .vt file with a non-MLP architecture.
    // Layer count 0 or a model with layernorm keyword should trigger rejection.
    let tmp_dir = std::env::temp_dir();
    let tmp_path = tmp_dir.join("volta_smoke_non_mlp.vt");
    std::fs::write(&tmp_path,
        "model m\n    layers 4 4\n    activation layernorm\n\ndataset d\n    type synthetic\n    batch 4\n\ntrain m on d\n    epochs 1\n    optimizer adam\n    lr 0.001\n    device cpu\n"
    ).expect("write temp vt");
    let out = std::process::Command::new(env!("CARGO_BIN_EXE_volta"))
        .args(["compile-train", tmp_path.to_str().unwrap()])
        .current_dir(env!("CARGO_MANIFEST_DIR"))
        .output()
        .expect("volta binary runnable");
    let _ = std::fs::remove_file(&tmp_path);
    // compile-train on an unsupported model must fail or warn
    // If it exits 0 but outputs an error/warning, that's also acceptable for now —
    // but non-zero exit is the target behavior (per 05-03 truth-pass).
    assert!(
        !out.status.success() || {
            let s = String::from_utf8_lossy(&out.stderr);
            s.contains("unsupported") || s.contains("MLP-only") || s.contains("not supported")
        },
        "expected compile-train to reject non-MLP model, got exit={} stderr={}",
        out.status,
        String::from_utf8_lossy(&out.stderr)
    );
}
```
